## 实现步骤

### Step 1：创建 `scripts/check_surface_naming.mjs`

脚本契约（严格遵守）：

- 入口：`node scripts/check_surface_naming.mjs`（无参数；CI & 本地同形态）
- 退出码：
  - `0`：全部合规
  - `1`：存在违规；stderr 按行输出 `<相对路径>:<行号>: forbidden token '<token>' found: "<上下文 60 字符>"`；stdout 输出总数摘要
- 规则表（硬编码在脚本顶部，显式列出）：

  ```js
  const RULES = [
    {
      id: "executive-no-gui-yin",
      includeGlobs: [
        "frontend/src/features/executive-dashboard/**/*.{ts,tsx,js,jsx}",
        "frontend/src/features/workbench/pages/DashboardPage.tsx",
      ],
      forbidden: ["归因"],
      allowedContains: [],            // 没有"以 XX 开头则豁免"的子串
      description: "executive 表层禁止出现'归因'（应使用'贡献拆解'）",
    },
    {
      id: "pnl-attribution-no-jing-ying-gong-xian",
      includeGlobs: ["frontend/src/features/pnl-attribution/**/*.{ts,tsx,js,jsx}"],
      forbidden: ["经营贡献"],
      allowedContains: [],
      description: "pnl-attribution 表层禁止出现'经营贡献'（应使用'归因'/'PnL 贡献'）",
    },
  ];
  ```

- 实现细节：
  - 用 `fs.readdirSync(dir, { withFileTypes: true, recursive: true })` 递归收集文件（Node 20+ 支持 `recursive`，Node 22 稳定；CI 用 22 可用）；**不**引入 glob 依赖
  - 对 glob pattern `**/*.{ts,tsx,js,jsx}` 用简单字符串匹配：收集目录下所有文件后按扩展名过滤 + 路径 startsWith 前缀匹配
  - 精确路径（如 `DashboardPage.tsx`）单独读文件
  - 逐行扫描（`split(/\r?\n/)`），记录匹配行号 + 前后 60 char 上下文
  - 命中任意一条规则 → exit(1)

- 代码骨架示例（非强制，subagent 可用等价实现）：

  ```js
  #!/usr/bin/env node
  // Surface naming lint — enforces §7.3 of the numeric-correctness design doc.
  import { readFileSync, readdirSync, statSync, existsSync } from "node:fs";
  import { join, relative, sep } from "node:path";
  import { fileURLToPath } from "node:url";

  const ROOT = fileURLToPath(new URL("..", import.meta.url));
  const EXTS = new Set([".ts", ".tsx", ".js", ".jsx"]);

  const RULES = [ /* see table above */ ];

  function listFilesRecursive(dir) {
    const out = [];
    if (!existsSync(dir)) return out;
    for (const ent of readdirSync(dir, { withFileTypes: true })) {
      const p = join(dir, ent.name);
      if (ent.isDirectory()) out.push(...listFilesRecursive(p));
      else out.push(p);
    }
    return out;
  }

  function collectTargets(rule) {
    const hits = new Set();
    for (const pattern of rule.includeGlobs) {
      if (pattern.endsWith(".tsx") || pattern.endsWith(".ts") || pattern.endsWith(".js") || pattern.endsWith(".jsx")) {
        // exact file
        const abs = join(ROOT, pattern);
        if (existsSync(abs)) hits.add(abs);
        continue;
      }
      // glob "frontend/src/features/<x>/**/*.{ts,tsx,...}"
      const baseDir = pattern.replace(/\/\*\*\/\*\.\{[^}]+\}$/, "");
      const absBase = join(ROOT, baseDir);
      for (const p of listFilesRecursive(absBase)) {
        if (EXTS.has(p.slice(p.lastIndexOf(".")))) hits.add(p);
      }
    }
    return [...hits];
  }

  let violations = 0;
  for (const rule of RULES) {
    const files = collectTargets(rule);
    for (const abs of files) {
      const text = readFileSync(abs, "utf8");
      const lines = text.split(/\r?\n/);
      for (let i = 0; i < lines.length; i += 1) {
        for (const tok of rule.forbidden) {
          const idx = lines[i].indexOf(tok);
          if (idx === -1) continue;
          const rel = relative(ROOT, abs).split(sep).join("/");
          const ctxStart = Math.max(0, idx - 30);
          const ctx = lines[i].slice(ctxStart, ctxStart + 60);
          process.stderr.write(`${rel}:${i + 1}: forbidden token '${tok}' (rule=${rule.id}): "${ctx}"\n`);
          violations += 1;
        }
      }
    }
  }

  if (violations > 0) {
    process.stderr.write(`\n${violations} surface-naming violation(s).\nRule descriptions:\n`);
    for (const r of RULES) process.stderr.write(`  - ${r.id}: ${r.description}\n`);
    process.exit(1);
  } else {
    process.stdout.write("surface-naming lint: OK\n");
  }
  ```

```yaml

    name: Frontend Lint
    runs-on: ubuntu-latest
    timeout-minutes: 5

    defaults:
      run:
        working-directory: frontend

    steps:
      - uses: actions/checkout@v4

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "22"
          cache: npm
          cache-dependency-path: frontend/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Surface naming lint
        working-directory: .
        run: node scripts/check_surface_naming.mjs

      - name: ESLint
        run: npx eslint .
```
