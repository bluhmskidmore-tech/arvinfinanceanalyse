## 背景上下文

### 当前 PnlAttributionSection.tsx 的 5 个根问题

1. **接口过时**：接 `data: PnlAttributionPayload` + `isLoading` + `isError`，现在 payload 字段是 Numeric → render 会挂
2. **两套反向 tone map**：
   - `accentMap.positive = "#2f8f63"`（绿）用于列表
   - `chartToneColor.positive = "#cf1322"`（红）用于环形图
   - **用户在同一视图里看到同一 segment 的两种反向色彩，明确是 P1-4 bug**
3. **`Math.abs(s.amount)` × 3 处**：环形图 slice value、`maxAbsAmount`、列表条 width —— P1-4 要求彻底去掉
4. **环形图把负值 slice 渲染成正向面积**：负贡献在构成图里看起来像正贡献，P1-4 语义错乱
5. **读 `segment.display_amount`**：该字段已在 W2.3 contracts.ts 删除，此处是死路径

### 重写后的组件接口

```tsx
type PnlAttributionSectionProps = {
  attribution: DashboardAdapterOutput["attribution"];  // { vm, state, meta }
  onRetry: () => void;
};
```

### 重写后的视觉设计（bipolar bar chart）

**左侧 ECharts**：改为 **horizontal bipolar bar chart**
- yAxis: category = segments label
- xAxis: value，中心 0 基线（value 域覆盖 [-maxAbs, +maxAbs]）
- 每个 segment 作为一个 bar：value = `segment.amount.raw`（**不取绝对值**）
- bar 颜色按 `toneFromNumeric(segment.amount)` 或 `TONE_COLOR[segment.tone]`
- tooltip 显示 `${segment.label}: ${segment.amount.display}`
- 图表标题 center 文字用 `attribution.vm.total.display`

**右侧列表条**：改为 **bipolar bar**
- 每行一个 segment：label + `amount.display`
- bar 容器：`display: flex`，左右两侧分别占 50% 宽度，0 基线居中
- 如果 `amount.raw > 0`：bar 在右侧，宽度 `(amount.raw / maxAbsAmount) * 50%`
- 如果 `amount.raw < 0`：bar 在左侧，宽度 `(|amount.raw| / maxAbsAmount) * 50%`
- 如果 `amount.raw = null`：bar 不渲染（仅显示 `amount.display = "—"`）
- bar 颜色按 `TONE_COLOR[segment.tone]`

### 色彩唯一出口

- 删除 `accentMap` 和 `chartToneColor` 两个 const
- 改用 `TONE_COLOR` from `../../../utils/tone`

### Step 2：完全重写 `PnlAttributionSection.tsx`

替换为以下内容：

```tsx
import { useMemo } from "react";

import ReactECharts from "../../../lib/echarts";
import { DataSection } from "../../../components/DataSection";
import { TONE_COLOR } from "../../../utils/tone";
import type { DashboardAdapterOutput } from "../adapters/executiveDashboardAdapter";
import {
  selectPnlMaxAbsAmount,
  selectPnlSegmentsForChart,
  selectPnlSegmentsForList,
  selectPnlTotal,
} from "../selectors/executiveDashboardSelectors";

type PnlAttributionSectionProps = {
  attribution: DashboardAdapterOutput["attribution"];
  onRetry: () => void;
};

const SECTION_EXTRA_STYLE = { color: "#5c6b82" } as const;

const CONTENT_ROW_STYLE = {
  display: "flex",
  gap: 20,
  alignItems: "flex-start",
  width: "100%",
} as const;

const CHART_CONTAINER_STYLE = { width: 240, height: 220, flexShrink: 0 } as const;

const LIST_COLUMN_STYLE = { display: "grid", gap: 14, flex: 1, minWidth: 0 } as const;

const ROW_HEADER_STYLE = {
  display: "flex",
  justifyContent: "space-between",
  marginBottom: 6,
  gap: 12,
} as const;

const BAR_TRACK_STYLE = {
  position: "relative" as const,
  height: 12,
  borderRadius: 999,
  background: "#ecf1f6",
  overflow: "hidden",
};

const BAR_CENTER_LINE_STYLE = {
  position: "absolute" as const,
  left: "50%",
  top: 0,
  bottom: 0,
  width: 1,
  background: "#c4cedc",
};

export default function PnlAttributionSection({
  attribution,
  onRetry,
}: PnlAttributionSectionProps) {
  const total = selectPnlTotal(attribution.vm);
  const chartSegments = selectPnlSegmentsForChart(attribution.vm);
  const listSegments = selectPnlSegmentsForList(attribution.vm);
  const maxAbs = selectPnlMaxAbsAmount(attribution.vm);

  const chartOption = useMemo(() => {
    if (chartSegments.length === 0) return null;
    // Labels in reverse so the first segment in data order renders at the top
    const reversed = [...chartSegments].reverse();
    return {
      tooltip: {
        trigger: "axis" as const,
        axisPointer: { type: "shadow" as const },
        formatter: (params: unknown) => {
          const entries = params as Array<{
            axisValue: string;
            data: { display?: string };
          }>;
          if (!entries || entries.length === 0) return "";
          const head = entries[0];
          return `${head?.axisValue ?? ""}<br/>${head?.data?.display ?? ""}`;
        },
      },
      grid: { left: 80, right: 16, top: 10, bottom: 10, containLabel: true },
      xAxis: {
        type: "value" as const,
        axisLine: { show: true, lineStyle: { color: "#c4cedc" } },
        splitLine: { lineStyle: { type: "dashed" as const, color: "#e4ebf5" } },
      },
      yAxis: {
        type: "category" as const,
        data: reversed.map((s) => s.label),
        axisTick: { show: false },
        axisLine: { show: false },
      },
      series: [
        {
          type: "bar" as const,
          data: reversed.map((s) => ({
            value: s.amount.raw ?? 0,
            display: s.amount.display,
            itemStyle: { color: TONE_COLOR[s.tone] },
          })),
          barWidth: 14,
          label: { show: false },
        },
      ],
    };
  }, [chartSegments]);

  const extra = total ? <span style={SECTION_EXTRA_STYLE}>{total.display}</span> : null;

  return (
    <DataSection
      title={attribution.vm?.title ?? "收益归因"}
      extra={extra}
      state={attribution.state}
      onRetry={onRetry}
    >
      <div style={CONTENT_ROW_STYLE}>
        {chartOption ? (
          <ReactECharts option={chartOption} style={CHART_CONTAINER_STYLE} />
        ) : null}
        <div style={LIST_COLUMN_STYLE}>
          {listSegments.map((segment) => (
            <div key={segment.id}>
              <div style={ROW_HEADER_STYLE}>
                <span>{segment.label}</span>
                <span style={{ color: TONE_COLOR[segment.tone], fontWeight: 600 }}>
                  {segment.amount.display}
                </span>
              </div>
              <div style={BAR_TRACK_STYLE}>
                <div style={BAR_CENTER_LINE_STYLE} />
                {renderBipolarFill({
                  raw: segment.amount.raw,
                  maxAbs,
                  color: TONE_COLOR[segment.tone],
                })}
              </div>
            </div>
          ))}
        </div>
      </div>
    </DataSection>
  );
}

function renderBipolarFill(opts: {
  raw: number | null;
  maxAbs: number;
  color: string;
}) {
  const { raw, maxAbs, color } = opts;
  if (raw === null || maxAbs === 0) return null;

  const widthPct = Math.min(100, Math.round((Math.abs(raw) / maxAbs) * 50));
  // raw > 0: bar grows from center to the right
  // raw < 0: bar grows from center to the left
  const style: React.CSSProperties = {
    position: "absolute",
    top: 0,
    bottom: 0,
    background: color,
    borderRadius: 999,
  };
  if (raw >= 0) {
    style.left = "50%";
    style.width = `${widthPct}%`;
  } else {
    style.right = "50%";
    style.width = `${widthPct}%`;
  }
  return <div style={style} />;
}
```

**关键特性：**
- `import React` 不需要（现代 Vite + React 17+ 自动 runtime），但 `React.CSSProperties` 要 `import type { CSSProperties } from "react"`。**注意：** 如果 `React.CSSProperties` typecheck 挂，改用 `import type { CSSProperties } from "react"` 然后本地用 `CSSProperties`。
- 整个文件**不出现** `Math.abs` 以外的 `Math.*`（`Math.abs` 用在 `renderBipolarFill` 内部 `Math.abs(raw)` 是必要的宽度计算）；**不出现**任何 `toFixed` / `/1e8` / `display_amount`
- **不出现** `accentMap` 或 `chartToneColor`
- 所有颜色从 `TONE_COLOR` 取
