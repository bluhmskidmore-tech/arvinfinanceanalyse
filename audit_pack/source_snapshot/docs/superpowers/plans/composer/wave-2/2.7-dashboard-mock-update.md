## 背景上下文

### 当前 `mocks/workbench.ts` 里数字字段（旧 shape）

```ts
// overviewPayload.metrics[i].value = "—"  (string)
// overviewPayload.metrics[i].delta = "—"  (string)
// pnlAttributionPayload.total = "—"  (string)
// pnlAttributionPayload.segments[i].amount = 0  (number)
// pnlAttributionPayload.segments[i].display_amount = "—"  (string)  ← 要删
// riskOverviewPayload.signals[i].value = "—"  (string)
// contributionPayload.rows[i].contribution = "—"  (string)
// contributionPayload.rows[i].completion = 0   (number, 保持不变)
```

### 升级目标

把每个 mock 数字字段替换成 `Numeric` 对象形式。因为 mock 是演示用的占位（文字是 "—"），所以 `Numeric.raw = null`、`display = "—"`。

**Numeric 构造模板：**

```ts
{ raw: null, unit: "yuan", display: "—", precision: 2, sign_aware: false }
```

具体每个字段的 `unit` / `sign_aware` 按 Numeric type 语义：

| 字段 | unit | sign_aware |
|---|---|---|
| `ExecutiveMetric.value`（金额） | `"yuan"` | `false`（规模不带符号） |
| `ExecutiveMetric.delta`（环比百分比） | `"pct"` | `true`（带符号变化） |
| `PnlAttributionPayload.total` | `"yuan"` | `true` |
| `AttributionSegment.amount` | `"yuan"` | `true` |
| `RiskSignal.value` | `"ratio"` | `false`（duration/%/dv01 统一用 ratio + display 自带单位） |
| `ContributionRow.contribution` | `"yuan"` | `true` |

### AttributionSegment 要**删除** `display_amount` 字段

W2.3 contracts.ts 已经把 `AttributionSegment` 改为 `{amount: Numeric}`，无 `display_amount`。Mock 里之前的 `display_amount: "—"` 必须删除，同时 `amount: 0` 升级为 `Numeric`。

// ------------------------------------------------------------------
// Wave 2.7 · mockClient.getOverview / getPnlAttribution 对拍
// ------------------------------------------------------------------

import { createApiClient } from "../../api/client";

describe("mockClient.getOverview · W2.7 contract", () => {
  it("returns envelope where every Numeric-shaped node passes isNumeric", async () => {
    const client = createApiClient({ mode: "mock" });
    const envelope = await client.getOverview();
    expect(envelope.result).toBeDefined();
    assertAllNumerics(envelope.result, "getOverview.result");
  });

  it("each metric has Numeric value and delta", async () => {
    const client = createApiClient({ mode: "mock" });
    const envelope = await client.getOverview();
    for (const metric of envelope.result.metrics) {
      expect(isNumeric(metric.value)).toBe(true);
      expect(isNumeric(metric.delta)).toBe(true);
    }
  });
});

describe("mockClient.getPnlAttribution · W2.7 contract", () => {
  it("returns envelope where every Numeric-shaped node passes isNumeric", async () => {
    const client = createApiClient({ mode: "mock" });
    const envelope = await client.getPnlAttribution();
    assertAllNumerics(envelope.result, "getPnlAttribution.result");
  });

  it("total and every segment.amount are Numeric", async () => {
    const client = createApiClient({ mode: "mock" });
    const envelope = await client.getPnlAttribution();
    expect(isNumeric(envelope.result.total)).toBe(true);
    for (const segment of envelope.result.segments) {
      expect(isNumeric(segment.amount)).toBe(true);
    }
  });

  it("segments no longer have display_amount field", async () => {
    const client = createApiClient({ mode: "mock" });
    const envelope = await client.getPnlAttribution();
    for (const segment of envelope.result.segments) {
      expect((segment as Record<string, unknown>).display_amount).toBeUndefined();
    }
  });
});
